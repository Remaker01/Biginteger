168 Biginteger Karatsuba(const Biginteger &a) const {
169 int half = (std::max(eff_len,a.eff_len) + 1) / 2;
170 Biginteger xh = getUpper(eff_len - half),xl = getLower(half); //第一个因数
171 Biginteger yh = a.getUpper(a.eff_len - half),yl = a.getLower(half); //第二个因数
172 //pl=xl * yl ph=xh * yh
173 Biginteger pl = xl.Multiply(yl);
174 Biginteger ph = xh.Multiply(yh);
175 //res= pl + ph * MODE_BASE^(half * 2) + ((xl + xh)(yl + yh) - pl - ph) * MODE_BASE(half)
176 return pl + ph.addZero(half * 2) + ((xl.Add(xh)).Multiply(yl.Add(yh)) - pl - ph).addZero(half);
177 }
